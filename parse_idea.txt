{
	t_list	*block; ---> inside CMD
	t_node	*pipe
	t_node	*cmd

	{
		add_tkn_to_block_list(tkn);
		tkn = tkn->next;
	}
	// tkn->content == '|'    OR    the last tkn

	cmd = malloc (t_node)
	cmd->block = block;
	if (tkn->next) // last round condition
	{
		pipe = malloc (t_node)
		pipe->left = cmd;
		pipe->right = parse(tkn->next)
	}
	if (pipe == NULL)
		return (block)
	return (pipe)
}
